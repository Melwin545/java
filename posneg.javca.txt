class Main {
    public static void main(String[] args) {

        int arr[] = {10, -5, 8, -12, 15, -3};

        int positive = 0, negative = 0;

        for(int i = 0; i < arr.length; i++) {
            if(arr[i] > 0)
                positive++;
            else if(arr[i] < 0)
                negative++;
        }

        System.out.println("Positive elements = " + positive);
        System.out.println("Negative elements = " + negative);
    }
}
